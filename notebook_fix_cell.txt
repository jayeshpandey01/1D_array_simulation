# This attempts to fix the PyTorch DLL loading issue

import sys
import os

# Fix 1: Add DLL paths to system PATH
python_dir = os.path.dirname(sys.executable)
dll_paths = [
    os.path.join(python_dir, 'Library', 'bin'),
    os.path.join(python_dir, 'Lib', 'site-packages', 'torch', 'lib'),
]

for path in dll_paths:
    if os.path.exists(path):
        os.add_dll_directory(path)  # Python 3.8+
        os.environ['PATH'] = path + os.pathsep + os.environ['PATH']
        print(f"Added DLL path: {path}")

# Fix 2: Try importing torch
try:
    import torch
    print(f"✓ PyTorch {torch.__version__} loaded successfully!")
    print(f"✓ Device: {'CUDA' if torch.cuda.is_available() else 'CPU'}")
except Exception as e:
    print(f"✗ PyTorch failed to load: {e}")
    print("\nYou need to install Visual C++ Redistributable:")
    print("Download: https://aka.ms/vs/17/release/vc_redist.x64.exe")
    print("\nAfter installing, restart Jupyter and try again.")
